\section{Montar o conjunto de funções de transição}

	\textbf{Gramática}\\
	$S::=<A> <S>|\varepsilon$\\
	$A::=$ se cond entao $<B> |$ enquanto cond$ \{<C>\}$\\
	$B::= \{<C>\} | \{<C>\} $ else $\{<C>\}$\\			
	$C::= <A> |$ exp\\
	
	\textbf{Fita}\\
	se cond entao $\{$ enquanto cond $\{$ exp $\}\}$ else $\{$ exp $\}$\\
						
	\textbf{Funções de Transição:}\\
	
		\underline{ Regras de expansão para os não terminais $S, A, B , C$ }
	\begin{enumerate}[]
		\item	$\delta(0,\varepsilon,S) = \{(0,A S), (0, \varepsilon)\}$
		\item	$\delta(0, \varepsilon, A) = \{(0,$ se cond então $B),(0,$ enquanto cond $\{C\})\}$
		\item	$\delta(0,\varepsilon,B) = \{ (0, \{C\}), (0, \{C\}$ else $\{C\})\}$
		\item	$\delta(0,\varepsilon,C) = \{ (0, A), (0,$ exp$)\}$		
	\end{enumerate}
	
		\underline{ Regras de casamento para os terminais se, cond, entao, enquanto, $\{, \}$, else, exp }
	\begin{enumerate}[start=5]

		\item 	$\delta(0, se, se) = \{(0, \varepsilon)\})$
		\item	$\delta(0, cond, cond) = \{(0, \varepsilon)\})$
		\item	$\delta(0, entao, entao) = \{(0, \varepsilon)\})$
		\item	$\delta(0, enquanto, enquanto) = \{(0, \varepsilon)\})$
		\item	$\delta(0, \{, \{) = \{(0, \varepsilon)\})$
		\item	$\delta(0, \}, \}) = \{(0, \varepsilon)\})$
		\item	$\delta(0, else, else) = \{(0, \varepsilon)\})$
		\item	$\delta(0, exp, exp) = \{(0, \varepsilon)\})$
	\end{enumerate}
